CREATE TABLE IF NOT EXISTS graavl_policy_metadata (
  policy_id         BIGSERIAL PRIMARY KEY,

  policy_key        VARCHAR(120) NOT NULL,           -- stable name like ISIN_LICENSED_BLOCK_ALICLOUD
  domain            VARCHAR(100) NOT NULL,           -- e.g. INSTRUMENT, ACCOUNT
  service_name      VARCHAR(120) NOT NULL,           -- e.g. dist-instrument-service

  priority          INTEGER NOT NULL CHECK (priority BETWEEN 1 AND 10),

  result_action     policy_action NOT NULL,          -- PASS/FILTER/FULL_MASK/PARTIAL_MASK
  policy_type       policy_type NOT NULL DEFAULT 'DATABASE',

  -- Destination scoping (AliCloud etc.)
  destination_type  destination_type NOT NULL DEFAULT 'ANY',
  destination_key   VARCHAR(120),                    -- e.g. 'ALICLOUD' when destination_type=CLOUD

  -- Optional json-logic for additional matching conditions
  json_logic        TEXT,

  enabled           BOOLEAN NOT NULL DEFAULT TRUE,
  description       TEXT,

  created_by        VARCHAR(80),
  created_at        TIMESTAMP NOT NULL DEFAULT now(),
  updated_at        TIMESTAMP
);

-- Uniqueness
CREATE UNIQUE INDEX IF NOT EXISTS ux_policy_key_scope
  ON graavl_policy_metadata(domain, service_name, destination_type, COALESCE(destination_key,''), policy_key);

-- Fast lookup
CREATE INDEX IF NOT EXISTS ix_policy_scope
  ON graavl_policy_metadata(domain, service_name, destination_type, destination_key, enabled, priority);


CREATE TABLE IF NOT EXISTS graavl_policy_criteria (
  criteria_id     BIGSERIAL PRIMARY KEY,

  policy_id       BIGINT NOT NULL REFERENCES graavl_policy_metadata(policy_id) ON DELETE CASCADE,

  criteria_key    VARCHAR(150) NOT NULL,           -- e.g. 'instrument.isin', 'account.number', 'distribution.cloud'
  operator        criteria_operator NOT NULL,      -- EQ/IN/LIKE/...
  source_type     criteria_source NOT NULL DEFAULT 'PAYLOAD',  -- PAYLOAD/HEADER/SYSTEM

  data_type       VARCHAR(30) DEFAULT 'STRING',    -- STRING/BOOLEAN/NUMBER/DATE (informational)
  required        BOOLEAN NOT NULL DEFAULT TRUE,   -- if true and extractor returns nothing -> policy cannot match

  created_at      TIMESTAMP NOT NULL DEFAULT now()
);

CREATE INDEX IF NOT EXISTS ix_criteria_policy
  ON graavl_policy_criteria(policy_id);

CREATE INDEX IF NOT EXISTS ix_criteria_key
  ON graavl_policy_criteria(criteria_key);
